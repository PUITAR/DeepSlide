\section{HAMMER Approach: Method Overview}%% ITEM 6 TYPE Frame
%% ITEM 6 TYPE Frame
\begin{frame}
\frametitle{HAMMER: Hamiltonian Curiosity Augmented LLM Reinforcement}

\begin{figure}[h]
\centering
\includegraphics[width=0.9\textwidth,height=0.7\textheight,keepaspectratio]{picture/overview.pdf}
\caption{HAMMER training schema overview}

\begin{itemize}
\item \textbf{HAMMER}: Two-component training schema for LLMs
\item \textbf{Component 1}: Semantic embedding via LLM forward propagation
\item \textbf{Component 2}: Hamiltonian curiosity order for training sequence
\item Goal: Transfer diversity metrics into dynamic reinforcement learning
\end{itemize}
\end{frame}%% ITEM 7 TYPE Frame
%% ITEM 7 TYPE Frame
\begin{frame}
\frametitle{Semantic Embedding Generation}

\begin{itemize}
\item Sentence embeddings derived from LLM's forward propagation
\item Captures the model's intrinsic semantic understanding
\item Each training sample $x_i$ produces embedding $e_i$
\item Pairwise similarities form complete graph $G = (V, E)$
\item Similarity matrix $S$ where $S_{ij} = \text{sim}(e_i, e_j)$
\end{itemize}

\vspace{0.3cm}
\textbf{Key Insight}: Embeddings reflect how the LLM itself perceives semantic relationships, not external metrics.
\end{frame}%% ITEM 8 TYPE Frame
%% ITEM 8 TYPE Frame
\begin{frame}
\frametitle{Hamiltonian Curiosity Order}

\begin{itemize}
\item Complete graph $G$ with $n$ vertices (samples)
\item Hamiltonian path visits each vertex exactly once
\item We find path with \textbf{minimum cumulative similarity}
\item Minimizes: $\sum_{k=1}^{n-1} S_{\pi(k),\pi(k+1)}$
\item Result: Curiosity-driven training sequence
\end{itemize}

\vspace{0.3cm}
\textbf{Why this works}: By minimizing similarity between consecutive samples, we maximize semantic diversity during training, retaining exploration in early phases.
\end{frame}%% ITEM 9 TYPE Frame
%% ITEM 9 TYPE Frame
\begin{frame}
\frametitle{Summary: HAMMER Methodology}

\begin{itemize}
\item \textbf{Core Idea}: Transfer diversity metrics into dynamic RL
\item \textbf{Step 1}: Generate semantic embeddings via LLM forward propagation
\item \textbf{Step 2}: Compute pairwise similarity matrix
\item \textbf{Step 3}: Find Hamiltonian path with minimum cumulative similarity
\item \textbf{Result}: Curiosity-driven training order retains exploration
\end{itemize}

\vspace{0.3cm}
\textbf{Key Benefit}: Prevents premature convergence, promotes better generalization in LLMs.
\end{frame}%% ITEM 10 TYPE Section

\section{Key Results/Contributions}%% ITEM 11 TYPE Frame
%% ITEM 11 TYPE Frame
\begin{frame}
\frametitle{Key Results: HAMMER's Consistent Performance Gains}

\begin{itemize}
\item \textbf{HAMMER achieves 3--4\% average accuracy improvement} across diverse mathematical reasoning benchmarks
\item Tested on 4 model families: Qwen3-1.7B, 4B, 8B, and Deepseek-R1-Distill-Llama3-8B
\item Two training methods: DAPO and GRPO reinforcement learning
\item Benchmarks: AIME 2024/2025, AMC 2023, OlympiadBench
\item Metrics: pass@1, pass@10, pass@k, cons@k, and average accuracy
\end{itemize}

\vspace{0.5cm}
\textbf{Core Finding:} Diversity-driven ordering (HAMMER) consistently outperforms random shuffle baseline by 3--4\% average accuracy.

\vspace{0.3cm}
\textbf{Why this matters:} Demonstrates stable convergence and stimulates model curiosity, avoiding local optimization pitfalls of difficulty-based curricula.
\end{frame}%% ITEM 12 TYPE Frame
