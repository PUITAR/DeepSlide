\begin{frame}
\frametitle{Introduction: Vector Embeddings and Similarity Search}

\begin{itemize}
    \item \textbf{Vector embeddings} convert words and sentences into numerical vectors
    \item Capture semantic meaning and relationships between concepts
    \item Example: $\heartsuit \overset{\text{embedding}}{\implies} [0.124, 1.564, ...]$
    \item AI systems need to find vectors most similar to query vectors
    \item This is called \textbf{Nearest Neighbor Search (NNS)}
\end{itemize}

\vspace{0.5cm}

\textbf{Key Insight}: Smaller distance = Higher similarity between vectors
\end{frame}%% ITEM 2 TYPE Frame

\begin{frame}
\begin{center}
\vspace{-0.5cm}
\color{white}\Large\textbf{Introduction to ANNS}
\end{center}

\vspace{0.5cm}

\begin{itemize}
    \item \textbf{Exact NNS} is computationally expensive for large datasets
    \item \textbf{Approximate Nearest Neighbor Search (ANNS)} finds close matches efficiently
    \item Mildly relaxes accuracy constraints for speed
    \item Essential for real-world AI applications:
    \begin{itemize}
        \item Information retrieval \cite{zhu2019accelerating}
        \item Pattern recognition \cite{kosuge2019object}
        \item Retrieval Augmented Generation (RAG) \cite{brown2020language}
    \end{itemize}
    \item Four main ANNS approaches: tree-based, hashing-based, quantization-based, and graph-based
\end{itemize}

\vspace{0.5cm}

\textbf{Graph-based approaches} offer high quality and low latency

\begin{frame}
{Graph-Based ANNS: Core Concepts}
\vspace{-0.2cm}
\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item \textbf{Graph Definition}: $G = (V, E)$ where $V$ are nodes (vectors) and $E$ are edges (connections)
\item \textbf{Six Degrees of Separation}: Any two nodes can be connected through at most 6 intermediate nodes
\item \textbf{Proximity Graphs}: Connect similar vectors to enable efficient search
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\begin{figure}
\centering
\includegraphics[width=0.9\textwidth,height=0.7\textheight,keepaspectratio]{picture/pg.jpg}
\caption{Proximity graph with 7 nodes and 7 edges}
\end{figure}
\end{column}
\end{columns}

\vspace{0.2cm}
\begin{itemize}
\item \textbf{Key Insight}: Graph structure enables finding nearest neighbors by traversing connections
\item \textbf{Challenge}: Balancing search speed vs. accuracy in high-dimensional spaces
\end{itemize}
\end{frame}%% ITEM 7 TYPE Frame

\begin{frame}
{Current Limitations \& Pain Points}
\vspace{-0.2cm}
\begin{itemize}
\item \textbf{Traditional Methods}: Struggle with computational efficiency during nearest neighbor searches
\item \textbf{Speed-Accuracy Trade-off}: Existing approaches can't optimize both simultaneously
\item \textbf{Graph Construction}: Most methods use both long and short edges:
\begin{itemize}
\item \textbf{Long edges}: For quick initial neighbor discovery (early stage)
\item \textbf{Short edges}: For precise late-stage refinement
\end{itemize}
\item \textbf{Fixed Parameters}: Current algorithms use uniform search strategies throughout the entire process
\item \textbf{Key Insight}: Search process should be divided into stages with different optimization strategies
\end{itemize}

\vspace{0.2cm}
\textbf{State-of-the-art}: HNSW \cite{malkov2018efficient} and other proximity graphs \cite{dobson2023scaling} show promise but need improvement in dynamic optimization.
\end{frame}%% ITEM 8 TYPE Section

\section{Methods: CSPG Approach}%% ITEM 12 TYPE Frame
%% ITEM 12 TYPE Frame
\begin{frame}
\begin{center}
\Large\textbf{CSPG: Cross-Partition Search on Proximity Graphs}
\end{center}

\vspace{5mm}

\begin{itemize}
    \item \textbf{Core Idea}: Partition dataset into smaller graphs to reduce distance computations
    \item \textbf{Key Innovation}: Two-stage search balancing speed and accuracy
    \item \textbf{Goal}: Faster search while maintaining high accuracy
    \item \textbf{Approach}: Random partitioning + shared routing vectors
\end{itemize}

\vspace{5mm}

\begin{center}
\textit{"When far from query: use large steps for speed}\\
\textit{When close to query: use small steps for accuracy"}
\end{center}
\end{frame}%% ITEM 13 TYPE Frame
%% ITEM 13 TYPE Frame

\Large\textbf{Partitioning Algorithm}
\end{center}

\vspace{3mm}

\begin{figure}[ht]

\begin{itemize}
    \item Randomly partition dataset $\mathcal{D}$ into multiple groups
    \item Build sparse, smaller graphs for each group ($\mathcal{G}_1$, $\mathcal{G}_2$)
    \item Shared routing vectors (red) connect different partitions
    \item Group-specific vectors: green for $\mathcal{G}_1$, blue for $\mathcal{G}_2$
\end{itemize}

\begin{frame}
\begin{center}
\Large\textbf{Two-Stage Search Algorithm}
\end{center}

\vspace{5mm}

\begin{itemize}
    \item \textbf{Stage 1: Fast Approaching}
    \begin{itemize}
        \item Greedy beam search using only one partition
        \item Small candidate set size $ef_1$ for speed
        \item Quickly zones into query's nearby region
    \end{itemize}
    
    \item \textbf{Stage 2: Precise Search}
    \begin{itemize}
        \item Cross-partition expansion with larger candidate set $ef_2$
        \item When encountering routing vectors, add all instances from all partitions
        \item Maximizes search space for final closest vectors
    \end{itemize}
    
    \item \textbf{Key Parameters}: $ef_1 < ef_2$ for optimal speed-accuracy balance
\end{itemize}
\end{frame}%% ITEM 15 TYPE Frame

\begin{frame}
{CSPG: Key Contributions}
\centering
\vspace{0.2cm}

\begin{itemize}
    \item \textbf{Novel ANNS Schema}: Crossing Sparse Proximity Graph (CSPG) framework
    \item \textbf{Random Partitioning}: Dataset split into groups with separate, sparser graphs
    \item \textbf{Routing Vectors}: Shared vectors enable cross-partition traversal
    \item \textbf{Two-Stage Search}: 
    \begin{itemize}
        \item[--] Fast greedy beam search for initial approach
        \item[--] Cross-partition expansion for precise results
    \end{itemize}
    \item \textbf{Key Advantage}: Achieves both higher speed \textbf{and} higher accuracy
\end{itemize}

\vspace{0.3cm}
\textbf{Core Innovation}: Random partitioning creates smaller, sparser graphs that enable faster traversal while maintaining accuracy through intelligent cross-partition search.
