\section{Takeaways and Advancements}%% ITEM 10 TYPE Frame
\begin{frame}
\deepslideid{S0007}

\frametitle{Multimodal Agentic Capabilities}

\begin{itemize}
\item \textbf{Unified vision-language foundation}: Early-fusion training aligns visual understanding with language reasoning
\item \textbf{Effective multimodal agentic workloads}: Handles complex tasks requiring multimodal inputs and multi-step planning
\item \textbf{Hybrid routing architecture}: Combines Gated Delta Networks with sparse MoE routing for efficiency
\item \textbf{Native support for long contexts}: Enables interaction with external tools under extended contexts
\end{itemize}
\end{frame}%% ITEM 11 TYPE Frame
\begin{frame}
\deepslideid{S0008}

\frametitle{Scaled RL Advancements}

\item \textbf{Progressively more complex task distributions} for robust generalization
\item \textbf{Improved agent performance} over prior Qwen models through scaled RL training
\item \textbf{Broad multilingual coverage}: Support for 201 languages and dialects
\end{itemize}
