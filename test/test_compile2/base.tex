%!TeX encoding = UTF-8
%!TeX program = xelatex

\documentclass[10pt]{beamer}
% Minimalist theme: using the 'default' theme as a base for minimalism
\usetheme{default}

%-------------------------------------------------------
% INCLUDE PACKAGES
%-------------------------------------------------------

\usepackage[utf8]{inputenc}
\usepackage[english]{babel}
\usepackage[T1]{fontenc}
\usepackage{helvet} % Clean sans-serif font for minimalism
\usepackage{booktabs} % For clean table lines

%-------------------------------------------------------
% DEFFINING AND REDEFINING COMMANDS
%-------------------------------------------------------

% colored hyperlinks - simplified for minimalism
\newcommand{\chref}[2]{
  \href{#1}{{\color{blue!70!black}#2}}
}

% Minimalist frame title style
\setbeamertemplate{frametitle}{
  \vspace{0.5em}
  \insertframetitle\par
  \vspace{0.2em}
  \rule{\textwidth}{0.4pt}
}

% Remove navigation symbols
\setbeamertemplate{navigation symbols}{}

% Simple itemize style
\setbeamertemplate{itemize items}[circle]
\setbeamertemplate{itemize subitem}[circle]

% Minimalist color theme
\setbeamercolor{structure}{fg=black}
\setbeamercolor{title}{fg=black}
\setbeamercolor{frametitle}{fg=black}
\setbeamercolor{normal text}{fg=black}
\setbeamercolor{item}{fg=black}

% Remove footline
\setbeamertemplate{footline}{}

% Clean block style
\setbeamertemplate{blocks}[rounded][shadow=false]
\setbeamercolor{block title}{fg=white,bg=gray!70!black}
\setbeamercolor{block body}{fg=black,bg=gray!10}

% Simple title page
\setbeamertemplate{title page}{
  \vspace{2cm}
  \begin{center}
    {\usebeamerfont{title}\usebeamercolor[fg]{title}\inserttitle}[1em]
    {\usebeamerfont{subtitle}\insertsubtitle}[2em]
    {\usebeamerfont{author}\insertauthor}[0.5em]
    {\usebeamerfont{institute}\insertinstitute}[1em]
    {\usebeamerfont{date}\insertdate}
  \end{center}
  \vfill
}

\input{title}

\begin{document}

\begin{frame}[noframenumbering,plain]
  \titlepage 
\end{frame}

% Minimalist table of contents
\begin{frame}{Contents}
  \tableofcontents
\end{frame}

\input{content}

% references - minimalist style
\begin{frame}[plain]
    \frametitle{References}
    \bibliographystyle{plain}
    \bibliography{ref}
\end{frame}

\end{document}
