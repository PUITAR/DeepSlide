%% ITEM 2 TYPE Frame
\begin{frame}
\begin{center}
\LARGE\textbf{CSPG: Cross-Partition Sparse Proximity Graph}
\end{center}

\vspace{0.5cm}

\begin{itemize}
    \item \textbf{Goal}: Improve graph-based ANNS with partitioning strategy
    \item \textbf{Key Insight}: Reduce distance computations while maintaining accuracy
    \item \textbf{Core Idea}: Random partitioning with shared routing vectors
    \item \textbf{Two Key Algorithms}: Partitioning algorithm and search algorithm
    \item \textbf{Approach}: Two-stage search combining speed and precision
\end{itemize}
\end{frame}%% ITEM 3 TYPE Frame
%% ITEM 3 TYPE Frame
\begin{frame}
\begin{center}
\Large\textbf{Partitioning Algorithm}
\end{center}

\vspace{0.3cm}

\begin{itemize}
    \item \textbf{Random Partitioning}: Dataset $\mathcal{D}$ divided into multiple groups
    \item \textbf{Shared Routing Vectors}: Common vectors connect different partitions
    \item \textbf{Sparse Subgraphs}: Each group builds its own sparse proximity graph
    \item \textbf{Key Advantage}: Sparser graphs allow larger steps for fast approach
    \item \textbf{Routing Mechanism}: Enables efficient cross-partition search
\end{itemize}

\vspace{0.3cm}
\textit{Example: Dataset divided into two groups with red routing vectors shared between them}
\end{frame}%% ITEM 4 TYPE Frame
%% ITEM 4 TYPE Frame
\begin{frame}
\begin{center}
\Large\textbf{Two-Stage Search Algorithm}
\end{center}

\vspace{0.3cm}

\begin{itemize}
    \item \textbf{Stage 1: Fast Approaching}: Greedy beam search on single partition
    \item \textbf{Stage 2: Precise Search}: Cross-partition expansion with routing vectors
    \item \textbf{Different Expansion Factors}: $ef_1 < ef_2$ for two stages
    \item \textbf{First Stage Goal}: Quickly zone into query's nearby region
    \item \textbf{Second Stage Goal}: Maximize search space across all partitions
\end{itemize}

\vspace{0.3cm}
\textit{Yellow arrows: First stage, Orange arrows: Second stage}
\end{frame}%% ITEM 5 TYPE Frame
